Lab1 Regression

2D Linear Regression

代码
%一元线性回归 使用BGD
x = load('ex1_1x.dat');
y = load('ex1_1y.dat');
figure;
plot(x,y,'o');
ylabel('Height in meters');
xlabel('Age in years');
m = length(y);
x = [ones(m,1),x];
theta = zeros(2,1);

function theta = bgd(x,y,alpha,iter,theta)
  for i=1:iter
    m = length(y);
    sum = zeros(2,1); %记录偏导
    H = x * theta;%变成一列
    
	%更新theta0 求和i=1:m (H-y) * 1
	for i=1:m
  		sum(1,1) = sum(1,1) + H(i)-y(i);
	end

	for i=1:m
  		sum(2,1) = sum(2,1) + (H(i)-y(i))*x(i,2);
	end
	
	theta = theta - (alpha * sum) / m;
  end
end

theta = bgd(x,y,0.07,1500,theta);

hold on; 
plot(x(:,2), x*theta, '-')
legend('Training data', 'Linear regression')
hold off;
