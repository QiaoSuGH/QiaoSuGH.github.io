2D Linear Regression and Understanding J(θ)

要求：

  • 记录经过一次迭代之后theta的值
  • 进行BGD，得到最终theta的值
  • 根据上一步theta的值进行预测
  • 作出J(θ)的图像
代码
%一元线性回归 使用BGD
x = load('ex1_1x.dat');
y = load('ex1_1y.dat');
figure;
plot(x,y,'o');
ylabel('Height in meters');
xlabel('Age in years');
m = length(y);
x = [ones(m,1),x];

theta = zeros(2,1);


function theta = bgd(x,y,num_iter,theta,alpha)
	m = length(y);
	n = length(x(1,:));
  %disp(n);
	iter = 0;
	while(true)
		sum = zeros(n,1);
    H = x * theta;
		for i=1:n
			for j=1:m %计算第i个偏导数--遍历样本点，求和
				sum(i,1) = sum(i,1) + (H(j)-y(j))*x(j,i);
			end
		end
		theta = theta - (alpha * sum) / m;
		iter = iter+1;
		if iter > num_iter || abs(theta)<1e-4
      disp(iter);
			break;
		end
	end
end


theta = bgd(x,y,1500,theta,0.07);

hold on; 
plot(x(:,2), x*theta, '-')
legend('Training data', 'Linear regression')


boy1 = [1 , 3.5];
boy2 = [1 , 7];
plot(boy1(:,2),boy1 * theta, 'sr');
plot(boy2(:,2),boy2 * theta, 'sr');


%作出J(theta)的图像

function jij = cal_j_vals(x,y,t) 
  m = length(y);
  jij = (x*t - y)' * (x*t - y) / (2 * m); 
end;

j_vals = zeros(100,100);

theta0_vals = linspace(-3,3,100);
theta1_vals = linspace(-1,1,100);
for i = 1:length(theta0_vals)
  for j = 1:length(theta1_vals)
    t = [theta0_vals(i); theta1_vals(j)];
    j_vals(i,j) = cal_j_vals(x,y,t);
  end;
end;
j_vals = j_vals';
figure;
surf(theta0_vals, theta1_vals,j_vals);
xlabel('\theta_0'); ylabel('\theta_1');
[X,Y] = meshgrid(theta0_vals,theta1_vals);

Multivariate Linear Regression

要求：

  • 调整不同的学习率，使得损失函数的曲线形状接近于给出的曲线
  • 比较不同的学习率得到的不同的损失函数曲线
代码
x = load('ex1_2x.dat');
y = load('ex1_2y.dat');
% x: living area and the number of bedrooms
% y: house price
m = length(y);
x = [ones(m,1),x];
sigma = std(x);
mu = mean(x);
% preprocessing
x(:,2) = (x(:,2) - mu(2)) ./ sigma(2);
x(:,3) = (x(:,3) - mu(3)) ./ sigma(3);
n = 3;
%disp(n);
theta = zeros(n,1)

function theta = bgd(x,y,alpha,iter,theta)
  for i=1:iter
    m = length(y);
    n = 3;
    sum = zeros(n,1); %记录偏导
    H = x * theta;%变成一列

    for j=1:n
      for i=1:m
        sum(j,1) = sum(j,1) + (H(i)-y(i))*x(i,j);
      end
    end
    theta = theta - (alpha*sum)/m;
  end
end

theta = zeros(n,1);
function val_of_j = cal_j_vals(x,y,theta) 
  m = length(y);
  val_of_j = (x*theta - y)' * (x*theta - y) / (2 * m); 
end;

alpha = 0.06;
J = zeros(50,1);

for num_iterations = 1:50
  J(num_iterations) = cal_j_vals(x,y,theta);
  theta = bgd(x,y,alpha,num_iterations,theta);
end;

figure;
p0 = plot(0:49,J(1:50),'-');
hold on;
xlabel('Numbber of iterations')
ylabel('Cost J')


%批量改变alpha
figure;
alpha = 0.01;
while alpha < 0.1
  J = zeros(50,1);
  theta = zeros(n,1);
  for num_iterations = 1:50
  J(num_iterations) = cal_j_vals(x,y,theta);
  theta = bgd(x,y,alpha,num_iterations,theta);
  end;
  plot(0:49,J(1:50),'-');
  hold on;
  alpha = alpha * 3;
end;

  • 如果学习率太小，例如0.001，那么损失函数下降得非常缓慢；
