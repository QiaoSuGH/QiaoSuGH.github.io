实验一：OpenGL基本使用

目标：自行配置OpenGL环境，熟悉使用OpenGL

要求：在屏幕上绘制出一个圆，然后绘制出圆的外接正方形，然后再绘制出正方形的外接圆，重复以上操作几次。

迭代几次，每次更新圆的半径即可。

#include<GL/glut.h>
#include <math.h>
const int n = 1000;
GLdouble R = 0.05f;
const GLdouble Pi = acos(-1);

void myDisplay(void)
{
	glClear(GL_COLOR_BUFFER_BIT);
	glBegin(GL_POINTS);
	GLdouble d = sqrt(2);

	int num_iter = 4;
	for (int i = 0; i < num_iter; ++i)
	{
		for (int j = 0; j < n; ++j)
			glVertex2f(R * cos(2 * Pi / n * j), R * sin(2 * Pi / n * j));
		GLdouble x1 = -R, y1 = R;
		while (x1 < R)
		{
			glVertex2f(x1, R), glVertex2f(x1, -R);
			glVertex2f(-R, y1),glVertex2f(R, y1);
			x1 += 0.001;y1 -= 0.001;
		}
		R *= d;
	}
	glEnd();
	glFlush();
}

int main(int argc, char** argv)
{
	glutInit(&argc, argv);
	glutInitDisplayMode(GLUT_RGB | GLUT_SINGLE);
	glutInitWindowSize(1000, 1000);
	glutInitWindowPosition(400, 100);
	
	glutCreateWindow("Lab01");
	glutDisplayFunc(&myDisplay);
	glutMainLoop();

	return 0;
}
