void Midpoint_Line(GLint x0, GLint y0, GLint x1, GLint y1)
{
	//确保x0 较小
	glClear(GL_COLOR_BUFFER_BIT);
	GLint t;
	if (x1 < x0)
	{
		t = x1;x1 = x0, x0 = t;
		t = y1;y1 = y0, y0 = t;
	}
	GLint a, b, d1, d2, d, x, y;
	x = x0, y = y0;
	a = y0 - y1, b = x1 - x0;
	if (a == 0)
	{//平行于x轴
		while (x <= x1){drawPiexl(x, y);x++;}
		return;
	}
	if (b == 0)
	{//平行于y轴
		if (y0 > y1) { t = y1; y1 = y0; y0 = t; }
		y = y0;
		while (y <= y1){drawPiexl(x, y);y++;}
		return;
	}

	int dx = x1 - x0;
	int dy = y1 - y0;
	double k = double(dy) / double(dx);
	printf("%f\n", k);
	if (abs(k) <= 1)
	{
		if (k > 0)
		{//一般情况
			d1 = 2 * a, d2 = 2 * (a + b);
			d = 2 * a + b;
			drawPiexl(x, y);
			while (x < x1)
			{
				if (d < 0) { x++, y++, d += d2; }
				else { x++, d += d1; }
				drawPiexl(x, y);
			}
		}
		else
		{
			d = 2 * a - b;
			d1 = a - b, d2 = a;
			drawPiexl(x, y);
			while (x < x1)
			{
				if (d > 0) { x++, y--, d += d1; }
				else { x++, d += d2; }
				drawPiexl(x, y);
			}
		}
	}
	else
	{
		if (k > 0)
		{
			d = a + 2 * b;
			d1 = b;//d < 0
			d2 = a + b;//d > 0
			while (y < y1)
			{
				if (d > 0) { y++, x++, d += d2; }
				else { y++, d += d1; }
				drawPiexl(x, y);
			}
		}
		else
		{
			d = a - 2 * b;
			d1 = a - b, d2 = -b;
			while (y > y1)
			{
				if (d < 0) { y--, x++, d += d1; }
				else { y--, d += d2; }
				drawPiexl(x, y);
			}
		}
	}

}


void Bresenham_Line(GLint x0, GLint y0, GLint x1, GLint y1)
{
	glClear(GL_COLOR_BUFFER_BIT);
	GLint t;
	if (x1 < x0)
	{
		t = x1;x1 = x0, x0 = t;
		t = y1;y1 = y0, y0 = t;
	}

	GLint x = x0, y = y0;
	if (x0 == x1)
	{//平行于y轴
		if (y0 > y1) { t = y1; y1 = y0; y0 = t; }
		while (y <= y1)
		{
			drawPiexl(x, y);
			y++;
		}
		return;
	}
	if (y0 == y1)
	{//平行于x轴
		while (x <= x1)
		{
			drawPiexl(x, y);
			x++;
		}
		return;
	}

	int dx = x1-x0, dy =y1-y0;
	double k = double(dy) / double(dx);
	GLdouble e = -0.5;
	GLint i = 0;
	if (abs(k) <= 1)
	{
		//处理k>0
		if (k > 0)
		{
			for (i = 0; i <= dx; ++i)
			{
				drawPiexl(x, y);
				x += 1, e += k;
				if (e >= 0) { y++; e = e - 1; }
			}
		}
		else
		{
			k = abs(k);
			for (i = 0; i <= dx; ++i)
			{
				drawPiexl(x, y);
				x += 1, e += k;
				if (e >= 0) { y--; e = e - 1; }
			}
		}
	}
	else
	{
		if (k > 0)
		{
			k = 1.0 / k;
			for (i = 0; i <= dy; ++i)
			{
				drawPiexl(x, y);
				y += 1, e += k;
				if (e >= 0) { x++; e = e - 1; }
			}
		}
		else
		{
			k = 1.0 / abs(k);
			for (i = dy; i <= 0; ++i)
			{
				drawPiexl(x, y);
				y -= 1, e += k;
				if (e >= 0) { x++; e = e - 1; }
			}
		}
	}
}


void DDA_Line(int x0, int y0, int x1, int y1)
{
	glClear(GL_COLOR_BUFFER_BIT);
	int t;
	if (x1 < x0)
	{
		t = x1; x1 = x0; x0 = t;
		t = y1; y1 = y0; y0 = t;
	}
	int x;
	float dx, dy, y, k;
	dx = x1 - x0, dy = y1 - y0;
	k = dy / dx;
	y = y0;
	for (x = x0; x <= x1; x++)
	{
		drawPiexl(x, int(y + 0.5));
		y += k;
	}
}

void mymouse(GLint button, GLint state, GLint x,GLint y)
{
	if (button == GLUT_LEFT_BUTTON && state == GLUT_DOWN && flag == 1)
	{
		m0 = x;
		n0 = y;
		flag = 0;
	}
	else if (button == GLUT_LEFT_BUTTON && state == GLUT_UP && flag == 0)
	{
		m1 = x;
		n1 = y;
		DrawLine();
		glFlush();
		flag = 1;
	}
}

void movemouse(GLint x, GLint y)
{
	m1 = x, n1 = y;
	DrawLine();
	glFlush();
}

#include<gl/glut.h>
#include<cmath>
#include<cstdio>

GLint m0, n0, m1, n1;
GLdouble r;
bool flag = 1;

{
	glBegin(GL_POINTS);
	glVertex2i(x, y);
	glEnd();
}

void drawCirPoint(int x, int y,int x0,int y0)
{
	glBegin(GL_POINTS);
	glVertex2f(x + x0, y + y0);
	glVertex2f(y + x0, x + y0);
	glVertex2f(-x + x0, y + y0);
	glVertex2f(y + x0, -x + y0);
	glVertex2f(x + x0, -y + y0);
	glVertex2f(-y + x0, x + y0);
	glVertex2f(-x + x0, -y + y0);
	glVertex2f(-y + x0, -x + y0);
	glEnd();
}
void DrawCircle()
{
	glClear(GL_COLOR_BUFFER_BIT);
	int x = 0, y = r, x0 = m0, y0 = n0;
	double d = 1.25 - r;
	drawPiexl(x0, y0);
	while (x <= y)
	{
		if (d < 0)
			d += 2 * x + 3;
		else
		{
			d += 2 * (x - y) + 5;
			y--;
		}
		x++;
		drawCirPoint(x, y, x0, y0);
	}
}

void mycicle()
{
	DrawCircle();
	glFlush();
}

void mymouse(GLint button, GLint state, GLint x, GLint y)

}
